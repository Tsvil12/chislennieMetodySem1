# ============================================================================
# Строки 1-4: Импорт библиотек
# ============================================================================
import numpy as np
from scipy.optimize import minimize_scalar
import math

# ============================================================================
# Строки 6-8: Определение функции
# ============================================================================
def f(x):
    return math.cos(5 * x**2)

# ============================================================================
# Строки 10-16: Четвёртая производная (численное дифференцирование)
# Формула центральных разностей 4-го порядка: f⁽⁴⁾(x) ≈ (f(x-2h)-4f(x-h)+6f(x)-4f(x+h)+f(x+2h))/h⁴
# ============================================================================
def f_4th_derivative(x):
    h = 1e-5
    return (f(x - 2*h) - 4*f(x - h) + 6*f(x) - 4*f(x + h) + f(x + 2*h)) / (h**4)

# ============================================================================
# Строки 18-20: Параметры задачи
# ============================================================================
a = 0
b = 5
epsilon = 1e-6
n = 2                                      # Симпсон требует чётного n

# ============================================================================
# Строки 22-25: Поиск максимума |f⁽⁴⁾(x)|
# ============================================================================
result = minimize_scalar(lambda x: -abs(f_4th_derivative(x)), bounds=(a, b), method='bounded')
max_f_4th_derivative = -result.fun

# ============================================================================
# Строки 27-43: Выбор шага и вычисление интеграла
# Формула остатка Симпсона: |R(h)| = (b-a)/180 · M₄ · h⁴
# ============================================================================
while True:
    h = (b - a) / n
    x = np.linspace(a, b, n + 1)
    fx = [f(xi) for xi in x]
    
    R = -((b - a) / 180) * max_f_4th_derivative * h**4
    
    if abs(R) > epsilon / 2:
        n *= 2
        continue
    
    # Формула Симпсона: h/3·(y₀ + 4·Σy_нечёт + 2·Σy_чёт + y_n)
    I_simpson = (h / 3) * (fx[0] + 4 * sum(fx[1:-1:2]) + 2 * sum(fx[2:-2:2]) + fx[-1])
    break

print(f"Количество подотрезков: {n}")
print(f"Приближенное значение интеграла: {I_simpson:.6f}")
print(f"Остаточный член: {R}")
