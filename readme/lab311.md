# ============================================================================
# Строки 1-4: Импорт библиотек
# ============================================================================
import numpy as np
from scipy.optimize import minimize_scalar
import math

# ============================================================================
# Строки 6-8: Определение функции
# ============================================================================
def f(x):
    return math.cos(5 * x**2)

# ============================================================================
# Строки 10-14: Вторая производная для оценки остаточного члена
# f''(x) = -10·sin(5x²) - 100x²·cos(5x²)
# ============================================================================
def f_double_prime(x):
    return -10 * math.sin(5 * x**2) - 100 * x**2 * math.cos(5 * x**2)

# ============================================================================
# Строки 16-18: Параметры задачи
# ============================================================================
a = 0
b = 5
epsilon = 1e-6

# ============================================================================
# Строки 20-23: Поиск максимума |f''(x)| на [a,b]
# ============================================================================
result = minimize_scalar(lambda x: -abs(f_double_prime(x)), bounds=(a, b), method='bounded')
max_f_double_prime = -result.fun

# ============================================================================
# Строки 25-32: Выбор шага из условия |R(h)| ≤ ε/2
# Формула: |R(h)| = (b-a)/12 · M₂ · h²
# ============================================================================
n = 1
h = (b - a) / n
R = (b - a) / 12 * max_f_double_prime * h**2

while R > epsilon / 2:
    n += 1
    h = (b - a) / n
    R = (b - a) / 12 * max_f_double_prime * h**2

# ============================================================================
# Строки 34-38: Вычисление интеграла по формуле трапеций
# ============================================================================
x_values = [a + i * h for i in range(n + 1)]
f_values = [f(x) for x in x_values]
integral_approximation = (h / 2) * (f_values[0] + f_values[-1] + 2 * sum(f_values[1:-1]))

# ============================================================================
# Строки 40-43: Вывод результатов
# ============================================================================
print(f"Количество подотрезков: {n}")
print(f"Приближённое значение интеграла: {integral_approximation:.6f}")
print(f"Остаточный член: {R}")
