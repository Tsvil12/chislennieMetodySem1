# Лабораторная работа №2 (Часть 1): Интерполяция полиномом Лагранжа

## Задание
Из методички, стр. 15, задание 1:
> "Используя интерполяционный многочлен Лагранжа степени n, построить на отрезке [a, b] графики заданной функции y=f(x) и полинома Лагранжа y=L_n(x). Вывести величину теоретической и практической погрешностей."

**Вариант:** №6 из списка функций (стр. 15) — `cos(sin x²)`

**Параметры:**
- Функция: `f(x) = cos(sin²x)`
- Отрезок: `[0, π]`
- Количество узлов: `n = 10` (согласно требованию n ≥ 10)

---

## Теоретические основы (по методичке)

### Интерполяционный многочлен Лагранжа (стр. 11)

\[
L_n(x) = \sum_{i=0}^{n} y_i \prod_{\substack{j=0 \\ j \neq i}}^{n} \frac{x - x_j}{x_i - x_j}
\]

### Остаточный член (стр. 11)

\[
R_n(x) = \frac{f^{(n+1)}(\xi)}{(n+1)!} \prod_{i=0}^{n} (x - x_i), \quad \xi \in [\min x_i, \max x_i]
\]

### Оценка погрешности (стр. 11)

\[
|R_n(x)| \le \frac{M_{n+1}}{(n+1)!} \left| \prod_{i=0}^{n} (x - x_i) \right|, \quad M_{n+1} = \sup_{x \in [a,b]} |f^{(n+1)}(x)|
\]

---

## Полный код с разбором

```python
# ============================================================================
# Строки 1-3: Импорт библиотек
# ============================================================================
import numpy as np          # Для массивов и математических операций
import matplotlib.pyplot as plt  # Для построения графиков (задание 1, стр. 15)
import math                 # Для константы π

# ============================================================================
# Строки 5-6: Определение интерполируемой функции
# ============================================================================
def f(x):
    return np.cos(np.sin(x)**2)
# Соответствие: Вариант №6 из таблицы на стр. 15 — "cos (sin x2)"
# В коде реализовано как cos(sin²x)

# ============================================================================
# Строки 8-18: Интерполяционный многочлен Лагранжа
# Формула: стр. 11, (2.1) L_n(x) = Σ y_i · Π (x - x_j)/(x_i - x_j)
# ============================================================================
def lagrange(x, x_nodes, y_nodes):
    n = len(x_nodes)                     # Количество узлов (n+1 в терминах методички)
    result = 0                           # Начальное значение суммы Σ
    for i in range(n):                   # Внешний цикл: суммирование по i
        term = y_nodes[i]                # Начинаем с y_i
        for j in range(n):               # Внутренний цикл: вычисление произведения Π
            if i != j:                   # Условие j ≠ i из формулы
                term *= (x - x_nodes[j]) / (x_nodes[i] - x_nodes[j])  # Каждый множитель
        result += term                   # Добавляем слагаемое к сумме
    return result                        # Возвращаем L_n(x)

# ============================================================================
# Строки 20-38: Поиск максимума модуля производной M = sup|f⁽ⁿ⁺¹⁾(x)|
# Формула: стр. 11, M_{n+1} = sup|f^{(n+1)}(x)|
# ============================================================================
def find_max_M(n):
    # Вложенная функция для численного вычисления производной (конечные разности)
    def f_nth_derivative(x, n):
        h = 1e-5                          # Шаг для численного дифференцирования
        if n == 0:
            return f(x)                   # Нулевая производная — сама функция
        elif n == 1:
            # Центральная разность для первой производной: (f(x+h)-f(x-h))/(2h)
            return (f(x + h) - f(x - h)) / (2 * h)
        else:
            # Рекурсивное вычисление производных высших порядков
            return (f_nth_derivative(x + h, n - 1) - f_nth_derivative(x - h, n - 1)) / (2 * h)
    
    samples = 1000                        # Количество точек для поиска максимума
    x_samples = np.linspace(0, math.pi, samples)  # Равномерная сетка на [0, π]
    max_val = 0
    
    for x in x_samples:                   # Перебираем все точки
        try:
            deriv_val = abs(f_nth_derivative(x, n))  # |f⁽ⁿ⁾(x)|
            if deriv_val > max_val:
                max_val = deriv_val       # Обновляем максимум
        except:
            continue
    
    return max_val                        # Возвращаем M = sup|f⁽ⁿ⁾(x)|

# ============================================================================
# Строки 40-42: Параметры задачи
# ============================================================================
a, b = 0, math.pi                         # Отрезок интерполяции [a, b]
n = 10                                    # Количество узлов (n ≥ 10 по заданию)

# ============================================================================
# Строки 44-46: Построение узлов интерполяции
# ============================================================================
x_nodes = np.linspace(a, b, n)            # Равномерные узлы: x_i = a + i·h
y_nodes = f(x_nodes)                      # Значения функции в узлах: y_i = f(x_i)

# ============================================================================
# Строки 48-53: Вычисление интерполяции во всех точках для графика
# ============================================================================
x_plot = np.linspace(a, b, 200)           # Тонкая сетка для построения графика
y_true = f(x_plot)                        # Точные значения функции

y_lagr = []                               # Список для значений полинома Лагранжа
for x in x_plot:                          # Для каждой точки x на тонкой сетке
    y_lagr.append(lagrange(x, x_nodes, y_nodes))  # Вычисляем L_n(x)

# ============================================================================
# Строки 55-56: Практическая погрешность
# Формула: ε_практ = max|f(x) - L_n(x)|
# ============================================================================
errors = np.abs(y_true - y_lagr)          # |f(x) - L_n(x)| в каждой точке
max_error = np.max(errors)                # Максимальная практическая погрешность

# ============================================================================
# Строка 58: Вычисление M = max|f⁽ⁿ⁺¹⁾(x)|
# ============================================================================
M = find_max_M(n)                         # M_{n+1} = sup|f^{(n+1)}(x)|
print(f"Найденное M для n={n}: {M:.2f}")

# ============================================================================
# Строки 60-66: Теоретическая оценка погрешности
# Формула: |R_n(x)| ≤ M/(n+1)! · |Π(x - x_i)|
# ============================================================================
def theoretical_error(x):
    omega = 1.0
    for xi in x_nodes:                    # Вычисляем произведение Π(x - x_i)
        omega *= (x - xi)
    return abs(omega) * M / math.factorial(n)  # M/(n+1)! · |Π(x - x_i)|

y_theory = [theoretical_error(x) for x in x_plot]  # Теоретическая оценка в каждой точке
max_theory = np.max(y_theory)             # Максимальная теоретическая оценка

# ============================================================================
# Строки 68-78: Вывод результатов
# ============================================================================
print("="*50)
print("РЕЗУЛЬТАТЫ ИНТЕРПОЛЯЦИИ")
print("="*50)
print(f"Функция: f(x) = cos(sin²x)")
print(f"Отрезок: [0, π]")
print(f"Количество узлов: {n}")
print(f"Найденное M = {M:.2f}")
print(f"Макс. реальная ошибка: {max_error:.2e}")
print(f"Макс. теоретическая оценка: {max_theory:.2e}")
print("="*50)

# ============================================================================
# Строки 80-115: Построение графиков (4 графика согласно заданию 1, стр. 15)
# ============================================================================
fig, ((ax1, ax2), (ax3, ax4)) = plt.subplots(2, 2, figsize=(12, 10))

# График 1: Функция и интерполяционный многочлен
ax1.plot(x_plot, y_true, 'b-', linewidth=2, label='Точная функция')
ax1.plot(x_plot, y_lagr, 'r--', linewidth=2, label=f'Интерполяция L_{n}(x)')
ax1.plot(x_nodes, y_nodes, 'ko', markersize=6, label='Узлы интерполяции')
ax1.set_xlabel('x')
ax1.set_ylabel('y')
ax1.set_title('Функция и интерполяционный многочлен')
ax1.legend()
ax1.grid(True, alpha=0.3)

# График 2: Реальная ошибка интерполяции
ax2.plot(x_plot, errors, 'g-', linewidth=2)
ax2.fill_between(x_plot, 0, errors, alpha=0.3, color='green')
ax2.set_xlabel('x')
ax2.set_ylabel('Ошибка')
ax2.set_title(f'Реальная ошибка ε_практ = {max_error:.2e}')
ax2.grid(True, alpha=0.3)

# График 3: Проверка интерполяции в узлах (должна быть точность)
ax3.plot(x_nodes, y_nodes, 'bo', markersize=8, label='Точные значения')
ax3.plot(x_nodes, [lagrange(x, x_nodes, y_nodes) for x in x_nodes], 
         'rx', markersize=8, linewidth=2, label='Интерполяция в узлах')
for i in range(len(x_nodes)):
    ax3.plot([x_nodes[i], x_nodes[i]], 
             [y_nodes[i], lagrange(x_nodes[i], x_nodes, y_nodes)], 
             'k--', alpha=0.5)
ax3.set_xlabel('x')
ax3.set_ylabel('y')
ax3.set_title('Проверка: в узлах L_n(x_i) = f(x_i)')
ax3.legend()
ax3.grid(True, alpha=0.3)

# График 4: Сравнение теоретической и практической погрешности
ax4.plot(x_plot, errors, 'g-', linewidth=2, label='Реальная ошибка')
ax4.plot(x_plot, y_theory, 'm--', linewidth=2, label='Теоретическая оценка')
ax4.set_xlabel('x')
ax4.set_ylabel('Ошибка')
ax4.set_title('Сравнение с теорией')
ax4.legend()
ax4.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()  # Отображение графиков

# ============================================================================
# Строки 117-125: Таблица значений в узлах
# ============================================================================
print("\nТАБЛИЦА ЗНАЧЕНИЙ В УЗЛАХ:")
print("-"*40)
print("№ узла\t   x\t\t   f(x)")
print("-"*40)
for i in range(len(x_nodes)):
    print(f"{i:3}\t{x_nodes[i]:8.4f}\t{y_nodes[i]:12.6f}")
print("-"*40)
