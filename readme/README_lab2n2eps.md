# Лабораторная работа №2 (Часть 3): Подбор степени полинома по точности

##  Задание
Из методички, стр. 15, задание 2 (продолжение):
> "Найти значение функции y = f(x) в точке x = x0 с заданной точностью ε."

В данной реализации используется **автоматический подбор степени полинома** n до достижения требуемой точности.

**Параметры:**
- Функция: `f(x) = cos(sin²x)`
- Отрезок: `[0, π]`
- Точка: `x0 = 1.5`
- Точность: `ε = 1e-10`
- Начальная степень: `n = 2`


## Теоретические основы

### Критерий остановки (стр. 11-12, также стр. 23)

Итерационный процесс продолжается, пока:
\[
|f(x_0) - L_n(x_0)| > \varepsilon
\]

При достижении погрешности ≤ ε процесс останавливается.

### Полином Лагранжа общего вида (стр. 11)

\[
L_n(x) = \sum_{i=0}^{n} y_i \prod_{\substack{j=0 \\ j \neq i}}^{n} \frac{x - x_j}{x_i - x_j}
\]

### Равномерные узлы (стр. 10)

\[
x_i = a + i \cdot h, \quad h = \frac{b-a}{n}
\]

## Полный код с построчным разбором

```python
# ============================================================================
# Строки 1-3: Импорт библиотек
# ============================================================================
import numpy as np
import matplotlib.pyplot as plt

# ============================================================================
# Строки 5-6: Определение интерполируемой функции
# ============================================================================
def f(x):
    return np.cos(np.sin(x)**2)
# Соответствие: Вариант №6, стр. 15

# ============================================================================
# Строки 8-12: Параметры задачи
# ============================================================================
a = 0                      # Левая граница
b = np.pi                  # Правая граница
x0 = 1.5                   # Точка интерполяции
epsilon = 1e-10            # Заданная точность

print(f"Функция: y = cos(sin²x)")
print(f"Отрезок: [{a}, {b}]")
print(f"Точка x0 = {x0}")
print(f"Точность ε = {epsilon}\n")

# ============================================================================
# Строки 14-16: Функция для создания равномерных узлов
# Формула: x_i = a + i·h, i = 0..n-1
# ============================================================================
def lagrange_nodes(n, interval_start=a, interval_end=b):
    """Возвращает равномерно распределённые узлы Лагранжа."""
    return np.linspace(interval_start, interval_end, n)

# ============================================================================
# Строки 18-28: Классический интерполяционный полином Лагранжа
# Формула: L_n(x) = Σ y_i · Π (x - x_j)/(x_i - x_j)
# ============================================================================
def interpolate_lagrange(x_values, y_values, x):
    """Классический интерполяционный полином Лагранжа."""
    result = 0
    for i in range(len(y_values)):
        product = 1
        for j in range(len(y_values)):
            if i != j:                     # Условие j ≠ i
                product *= (x - x_values[j]) / (x_values[i] - x_values[j])
        result += y_values[i] * product
    return result

# ============================================================================
# Строки 30-60: Итерационный подбор степени полинома n
# Критерий остановки: |f(x0) - L_n(x0)| ≤ ε
# ============================================================================
n = 2                      # Начальная степень полинома
iteration = 0
max_iterations = 10        # Ограничитель для предотвращения бесконечного цикла

while True:
    iteration += 1
    
    # Создаём n равномерных узлов
    x_nodes = lagrange_nodes(n)
    
    # Вычисляем значения функции в узлах
    y_nodes = [f(xi) for xi in x_nodes]
    
    print(f"\nИтерация №{iteration}, n={n}:")
    print("Узлы Лагранжа:", ", ".join(map(lambda x: f"{x:.6f}", x_nodes)))
    print("Значения функции в узлах:", ", ".join(map(lambda y: f"{y:.6f}", y_nodes)))
    
    # Точное значение в точке x0
    exact_value = f(x0)
    
    # Приближённое значение по Лагранжу
    approx_value = interpolate_lagrange(x_nodes, y_nodes, x0)
    
    # Вычисляем погрешность
    error = abs(exact_value - approx_value)
    
    print(f"Приближенное значение в x0: {approx_value:.6f}")
    print(f"Ошибка: {error:.6f}")
    
    # Проверка условия остановки
    if iteration >= max_iterations:
        print(f"Достигнуто максимальное число итераций ({max_iterations})")
        break
        
    # Если точность недостаточна, увеличиваем порядок полинома
    if error <= epsilon:
        print(f"Требуемая точность {epsilon} достигнута!")
        break
    else:
        n += 1

# ============================================================================
# Строки 62-76: Построение итогового графика
# ============================================================================
x_plot = np.linspace(a, b, 100)
y_exact = f(x_plot)
y_interpolated = [interpolate_lagrange(x_nodes, y_nodes, x) for x in x_plot]

plt.figure(figsize=(10, 6))
plt.plot(x_plot, y_exact, 'b-', linewidth=2, label='Точная функция')
plt.plot(x_plot, y_interpolated, 'r--', linewidth=2, label=f'Полином Лагранжа (n={n})')
plt.scatter(x_nodes, y_nodes, color='green', s=80, zorder=5, label='Узлы Лагранжа')
plt.scatter([x0], [exact_value], color='black', s=100, zorder=5, label=f'x0 = {x0}')
plt.axvline(x=x0, color='gray', linestyle=':', alpha=0.7)

plt.title('Интерполяция полиномом Лагранжа')
plt.xlabel('x')
plt.ylabel('y')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()
