# ============================================================================
# Строки 1-4: Импорт библиотек
# ============================================================================
import numpy as np
from scipy.optimize import minimize_scalar
import math

# ============================================================================
# Строки 6-8: Определение функции
# ============================================================================
def f(x):
    return math.cos(5 * x**2)

# ============================================================================
# Строки 10-14: Вторая производная
# ============================================================================
def f_2nd_derivative(x):
    return -10 * math.sin(5 * x**2) - 100 * x**2 * math.cos(5 * x**2)

# ============================================================================
# Строки 16-18: Параметры задачи
# ============================================================================
a = 0
b = 5
epsilon = 1e-6
n = 1

# ============================================================================
# Строки 20-23: Поиск максимума |f''(x)|
# ============================================================================
result = minimize_scalar(lambda x: -abs(f_2nd_derivative(x)), bounds=(a, b), method='bounded')
max_f_2nd_derivative = -result.fun

# ============================================================================
# Строки 25-36: Выбор шага и вычисление интеграла
# Остаток для прямоугольников: |R| = (b-a)³/(24·n²)·M₂
# Формула левых прямоугольников: ∫f(x)dx ≈ h·Σf(x_i), i=0..n-1
# ============================================================================
while True:
    h = (b - a) / n
    R = -(b - a)**3 / (24 * n**2) * max_f_2nd_derivative
    
    if abs(R) > epsilon / 2:
        n += 1
        continue
    
    x = np.linspace(a, b - h, n)          # Левые границы подынтервалов
    fx = [f(xi) for xi in x]
    I_rect = sum(fx) * h                  # h · Σf(x_i)
    break

print(f"Количество подотрезков: {n}")
print(f"Приближенное значение интеграла: {I_rect}")
print(f"Остаточный член: {R}")
